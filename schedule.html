        async function fetchBusETA(stopId, route) {

            // Note: In real production, use your own proxy to avoid CORS issues

            const url = `https://data.etabus.gov.hk/v1/transport/kmb/stop-eta/${stopId}`;

            try {

                const response = await fetch(url);

                const data = await response.json();

                return data.data.filter(item => item.route === route);

            } catch (e) { return []; }

        }

        function calculateMinutes(etaTime) {

            if (!etaTime) return "--";

            const diff = new Date(etaTime) - new Date();

            const mins = Math.floor(diff / 1000 / 60);

            return mins > 0 ? mins : "即將抵達";

        }
